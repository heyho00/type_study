다양한 타입이 들어올 수 있게 하려면 union type

let union: string | number = 123;


타입은 변수에 담아 쓸 수 있음.
type Name = string | number; //보통 대문자로 지정해 일반변수와 구분한다.
let 네임: Name = 123;

function 함수(x: number): number {
  //변수에 타입지정도 가능, 어떤 타입이 리턴될지도 지정가능!(뒤에꺼)
  return x * 2;
}
함수("123"); //문자열이 들어가면 에러가난다.
